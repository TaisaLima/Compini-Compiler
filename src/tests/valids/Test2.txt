PAR {
    SEQ {
        input numeroFatorial;
        var resultadoFat = 1;
        var i = 2;
        while (i <= numeroFatorial) {
            resultadoFat = resultadoFat * i;
            i = i + 1;
        }
        print("Fatorial de " + numeroFatorial + " = " + resultadoFat);
    }

    SEQ {
        input nFib;
        var a = 0;
        var b = 1;
        var c;
        var j = 0;
        while (j < nFib) {
            print("Fibonacci[" + j + "] = " + a);
            c = a + b;
            a = b;
            b = c;
            j = j + 1;
        }
    }
}
